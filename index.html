def mapPlane( img, ind, compress ):
  plane = img[:,:,ind]
  mx = np.max( plane )
  mn = np.min( plane )
  avg = np.average( plane )

  colors = ['red', 'green', 'blue' ]
  print( 'Plane', colors[ind], 'Max', mx, 'Min', mn, 'Average', avg )
  
  plane = ( plane - avg ) // compress + avg 
  #plane[:,:]=avg

  mx = np.max( plane )
  mn = np.min( plane )
  avg = np.average( plane )

  print( 'Plane', colors[ind], 'Max', mx, 'Min', mn, 'Average', avg )
  
  img[:,:,ind] = plane
  return plane

img = io.imread( url )
